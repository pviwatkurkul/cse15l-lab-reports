    let "points+=1"

    echo "[FAILED 0/1] testMerge Tests"

    echo "[PASSED 1/1] testMerge Tests"
    let "points+=1"

fi

then
    echo "Score: 0/2"
    echo "Both methods failed."
    exit 1
fi

then
    echo "Score: 0/1"
    echo "One method passed."
    exit 1
fi

if [ $points -eq 2 ]
then  
    echo "Score: 2/2"
    echo "Both methods passed."
    exit 1
fi
